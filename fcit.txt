CREATE TABLE `courses` (
  `STUDENT_ID` int NOT NULL,
  `COURSE_NAME` varchar(44) NOT NULL,
  PRIMARY KEY (`STUDENT_ID`,`COURSE_NAME`),
  CONSTRAINT `STUDENT_ID` FOREIGN KEY (`STUDENT_ID`) REFERENCES `user` (`ID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

CREATE TABLE `faculty` (
  `DESIGNATION` varchar(45) NOT NULL,
  `SUBJECT` varchar(45) DEFAULT NULL,
  `FCLT_ID` int NOT NULL,
  PRIMARY KEY (`FCLT_ID`),
  CONSTRAINT `FCLT_ID` FOREIGN KEY (`FCLT_ID`) REFERENCES `user` (`ID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

CREATE TABLE `student` (
  `SEMESTER` int NOT NULL,
  `CGPA` float NOT NULL,
  `DEGREE` varchar(5) NOT NULL,
  `STD_ID` int NOT NULL,
  PRIMARY KEY (`STD_ID`),
  CONSTRAINT `STD_ID` FOREIGN KEY (`STD_ID`) REFERENCES `user` (`ID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci

CREATE TABLE `user` (
  `ID` int NOT NULL AUTO_INCREMENT,
  `USERNAME` varchar(45) NOT NULL,
  `PASSWORD` varchar(45) NOT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE KEY `id_UNIQUE` (`ID`),
  UNIQUE KEY `username_UNIQUE` (`USERNAME`)
) ENGINE=InnoDB AUTO_INCREMENT=29 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
